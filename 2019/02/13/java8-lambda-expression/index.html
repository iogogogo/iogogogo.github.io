2014е№ҙ3жңҲеҸ‘еёғзҡ„Java 8пјҢжңүеҸҜиғҪжҳҜJavaзүҲжң¬жӣҙж–°дёӯеҸҳеҢ–жңҖеӨ§зҡ„дёҖж¬ЎгҖӮж–°зҡ„Java 8дёәејҖеҸ‘иҖ…еёҰжқҘдәҶи®ёеӨҡйҮҚйҮҸзә§зҡ„ж–°зү№жҖ§пјҢеҢ…жӢ¬LambdaиЎЁиҫҫејҸпјҢжөҒејҸж•°жҚ®еӨ„зҗҶпјҢж–°зҡ„Optionalзұ»пјҢж–°зҡ„ж—Ҙжңҹе’Ңж—¶й—ҙAPIзӯүгҖӮиҝҷдәӣж–°зү№жҖ§з»ҷJavaејҖеҸ‘иҖ…еёҰжқҘдәҶзҰҸйҹіпјҢзү№еҲ«жҳҜLambdaиЎЁиҫҫејҸзҡ„ж”ҜжҢҒпјҢдҪҝзЁӢеәҸи®ҫи®ЎжӣҙеҠ з®ҖеҢ–гҖӮжң¬зҜҮж–Үз« е°Ҷи®Ёи®әиЎҢдёәеҸӮж•°еҢ–пјҢLambdaиЎЁиҫҫејҸпјҢеҮҪж•°ејҸжҺҘеҸЈзӯүзү№жҖ§гҖӮ

иЎҢдёәеҸӮж•°еҢ–

еңЁиҪҜд»¶ејҖеҸ‘зҡ„иҝҮзЁӢдёӯпјҢејҖеҸ‘дәәе‘ҳеҸҜиғҪдјҡйҒҮеҲ°йў‘з№Ғзҡ„йңҖжұӮеҸҳжӣҙпјҢдҪҝд»–д»¬дёҚж–ӯең°дҝ®ж”№зЁӢеәҸд»Ҙеә”еҜ№иҝҷдәӣеҸҳеҢ–зҡ„йңҖжұӮпјҢеҜјиҮҙйЎ№зӣ®иҝӣеәҰзј“ж…ўз”ҡиҮійЎ№зӣ®е»¶жңҹгҖӮиЎҢдёәеҸӮж•°еҢ–е°ұжҳҜдёҖз§ҚеҸҜд»Ҙеё®еҠ©дҪ еә”еҜ№йў‘з№ҒйңҖжұӮеҸҳжӣҙзҡ„ејҖеҸ‘жЁЎејҸпјҢз®ҖеҚ•зҡ„иҜҙпјҢе°ұжҳҜйў„е…Ҳе®ҡд№үдёҖдёӘд»Јз Ғеқ—иҖҢдёҚеҺ»жү§иЎҢе®ғпјҢжҠҠе®ғеҪ“еҒҡеҸӮж•°дј йҖ’з»ҷеҸҰдёҖдёӘж–№жі•пјҢиҝҷж ·пјҢиҝҷдёӘж–№жі•зҡ„иЎҢдёәе°ұиў«иҝҷж®өд»Јз Ғеқ—еҸӮж•°еҢ–дәҶгҖӮ

дёәдәҶж–№дҫҝзҗҶи§ЈпјҢжҲ‘д»¬йҖҡиҝҮдёҖдёӘдҫӢеӯҗжқҘи®Іи§ЈиЎҢдёәеҸӮж•°еҢ–зҡ„дҪҝз”ЁгҖӮеҒҮи®ҫжҲ‘д»¬жӯЈеңЁејҖеҸ‘дёҖдёӘеӣҫд№Ұз®ЎзҗҶзі»з»ҹпјҢйңҖжұӮжҳҜиҰҒеҜ№еӣҫд№Ұзҡ„дҪңиҖ…иҝӣиЎҢиҝҮж»ӨпјҢзӯӣйҖүеҮәжҢҮе®ҡдҪңиҖ…зҡ„д№ҰзұҚгҖӮжҜ”иҫғеёёи§Ғзҡ„еҒҡжі•е°ұжҳҜзј–еҶҷдёҖдёӘж–№жі•пјҢжҠҠдҪңиҖ…еҪ“жҲҗж–№жі•зҡ„еҸӮж•°пјҡ

public List<Book> filterByAuthor(List<Book> books, String author) {
    List<Book> result = new ArrayList<>();
    for (Book book : books) {
        if (author.equals(book.getAuthor())) {
            result.add(book);
        }
    }
    return result;
}

зҺ°еңЁе®ўжҲ·йңҖиҰҒеҸҳжӣҙйңҖжұӮпјҢж·»еҠ иҝҮж»ӨжқЎд»¶пјҢжҢүз…§еҮәзүҲзӨҫиҝҮж»ӨпјҢдәҺжҳҜжҲ‘д»¬дёҚеҫ—дёҚеҶҚж¬Ўзј–еҶҷдёҖдёӘж–№жі•пјҡ

public List<Book> filterByPublisher(List<Book> books, String publisher) {
    List<Book> result = new ArrayList<>();
    for (Book book : books) {
        if (publisher.equals(book.getPublisher())) {
            result.add(book);
        }
    }
    return result;
}

дёӨдёӘж–№жі•йҷӨдәҶеҗҚз§°д№ӢеӨ–пјҢеҶ…йғЁзҡ„е®һзҺ°йҖ»иҫ‘еҮ д№ҺдёҖжЁЎдёҖж ·пјҢе”ҜдёҖзҡ„еҢәеҲ«е°ұжҳҜifеҲӨж–ӯжқЎд»¶пјҢеүҚиҖ…еҲӨж–ӯзҡ„жҳҜдҪңиҖ…пјҢеҗҺиҖ…еҲӨж–ӯзҡ„жҳҜеҮәзүҲзӨҫгҖӮеҰӮжһңзҺ°еңЁе®ўжҲ·еҸҲиҰҒеўһеҠ йңҖжұӮпјҢйңҖиҰҒжҢүз…§еӣҫд№Ұзҡ„е”®д»·иҝҮж»ӨпјҢжҳҜдёҚжҳҜйңҖиҰҒеҶҚж¬Ўе°ҶдёҠйқўзҡ„ж–№жі•еӨҚеҲ¶дёҖйҒҚпјҢе°ҶifеҲӨж–ӯжқЎд»¶ж”№дёәе”®д»·пјҹ No! иҝҷз§ҚеҒҡжі•иҝқиғҢдәҶDRYпјҲDonвҖҷt Repeat YourselfпјҢдёҚиҰҒйҮҚеӨҚиҮӘе·ұпјүеҺҹеҲҷпјҢиҖҢдё”дёҚеҲ©дәҺеҗҺжңҹз»ҙжҠӨпјҢеҰӮжһңйңҖиҰҒж”№еҸҳж–№жі•еҶ…йғЁйҒҚеҺҶж–№ејҸжқҘжҸҗй«ҳжҖ§иғҪпјҢж„Ҹе‘ізқҖжҜҸдёӘfilterByXxx()ж–№жі•йғҪйңҖиҰҒдҝ®ж”№пјҢе·ҘдҪңйҮҸеӨӘеӨ§гҖӮ

дёҖз§ҚеҸҜиЎҢзҡ„еҠһжі•жҳҜеҜ№иҝҮж»Өзҡ„жқЎд»¶еҒҡжӣҙй«ҳеұӮзҡ„жҠҪиұЎпјҢиҝҮж»Өзҡ„жқЎд»¶ж— йқһе°ұжҳҜеӣҫд№Ұзҡ„жҹҗдәӣеұһжҖ§пјҲжҜ”еҰӮд»·ж јгҖҒеҮәзүҲзӨҫгҖҒеҮәзүҲж—ҘжңҹгҖҒдҪңиҖ…зӯүпјүпјҢеҸҜд»ҘеЈ°жҳҺдёҖдёӘжҺҘеҸЈз”ЁдәҺеҜ№иҝҮж»ӨжқЎд»¶е»әжЁЎпјҡ

public interface BookPredicate {
    public boolean test(Book book);
}

BookPredicateжҺҘеҸЈеҸӘжңүдёҖдёӘжҠҪиұЎж–№жі•test()пјҢиҜҘж–№жі•жҺҘеҸ—дёҖдёӘBookзұ»еһӢеҸӮж•°пјҢиҝ”еӣһдёҖдёӘbooleanеҖјпјҢеҸҜд»Ҙз”Ёе®ғжқҘиЎЁзӨәеӣҫд№Ұзҡ„дёҚеҗҢиҝҮж»ӨжқЎд»¶гҖӮ

жҺҘдёӢжқҘжҲ‘д»¬еҜ№д№ӢеүҚзҡ„иҝҮж»Өж–№жі•иҝӣиЎҢйҮҚжһ„пјҢе°ҶfilterByXxx()ж–№жі•зҡ„з¬¬дәҢдёӘеҸӮж•°жҚўжҲҗдёҠйқўе®ҡд№үзҡ„жҺҘеҸЈпјҡ

public List<Book> filter(List<Book> books, BookPredicate bookPredicate) {
    List<Book> result = new ArrayList<>();
    for (Book book : books) {
        if (bookPredicate.test(book)) {
            result.add(book);
        }
    }
    return result;
}

е°ҶиҝҮж»Өзҡ„жқЎд»¶жҚўжҲҗBookPredicateзҡ„е®һзҺ°зұ»пјҢиҝҷйҮҢйҮҮз”ЁдәҶеҶ…йғЁзұ»пјҡ

// ж №жҚ®дҪңиҖ…иҝҮж»Ө
final String author = "еј дёү";
List<Book> result = filter(books, new BookPredicate() {
    @Override
    public boolean test(Book book) {
        return author.equals(book.getAuthor());
    }
});

// ж №жҚ®еӣҫд№Ұд»·ж јиҝҮж»Ө
final double price = 100.00D;
List<Book> result = filter(books, new BookPredicate() {
    @Override
    public boolean test(Book book) {
        return price > book.getPrice();
    }
});

йҮҚжһ„еүҚеҗҺжңүд»Җд№ҲеҢәеҲ«пјҹжҲ‘д»¬е°Ҷж–№жі•дёӯзҡ„ifеҲӨж–ӯжқЎд»¶жҚўжҲҗдәҶBookPredicateжҺҘеҸЈе®ҡд№үзҡ„test()ж–№жі•пјҢз”ЁдәҺеҲӨж–ӯжҳҜеҗҰж»Ўи¶іиҝҮж»ӨжқЎд»¶пјҢе°Ҷеӣҫд№ҰиҝҮж»Өзҡ„йҖ»иҫ‘дәӨз»ҷдәҶBookPredicateжҺҘеҸЈзҡ„е®һзҺ°зұ»пјҢиҖҢдёҚжҳҜеңЁfilter()ж–№жі•еҶ…йғЁе®һзҺ°иҝҮж»ӨпјҢиҖҢBookPredicateжҺҘеҸЈеҸҲжҳҜfilter()ж–№жі•зҡ„еҸӮж•°гҖӮд»ҘдёҠзҡ„жӯҘйӘӨпјҢе°ұжҳҜе°ҶиЎҢдёәеҸӮж•°еҢ–пјҢд№ҹе°ұжҳҜе°Ҷеӣҫд№ҰиҝҮж»Өзҡ„иЎҢдёәпјҲBookPredicateжҺҘеҸЈзҡ„е®һзҺ°зұ»пјүеҪ“еҒҡfilter()ж–№жі•зҡ„еҸӮж•°гҖӮзҺ°еңЁпјҢеҸҜд»ҘеҲ жҺүжүҖжңүfilterByXxx()зҡ„ж–№жі•пјҢеҸӘдҝқз•ҷfilter()ж–№жі•пјҢе°ұз®—еҗҺжңҹж•°жҚ®и§„жЁЎеҫҲеәһеӨ§пјҢйңҖиҰҒж”№еҸҳйӣҶеҗҲзҡ„йҒҚеҺҶж–№ејҸжқҘжҸҗй«ҳжҖ§иғҪпјҢеҸӘйңҖиҰҒеңЁfilter()ж–№жі•еҶ…йғЁеҒҡеҮәзӣёеә”зҡ„дҝ®ж”№пјҢиҖҢдёҚз”ЁеҺ»дҝ®ж”№е…¶д»–дёҡеҠЎд»Јз ҒгҖӮ

дёҚиҝҮпјҢBookPredicateжҺҘеҸЈеҸӘжҳҜй’ҲеҜ№еӣҫд№Ұзҡ„иҝҮж»ӨпјҢеҰӮжһңйңҖиҰҒеҜ№е…¶д»–еҜ№иұЎйӣҶеҗҲжҺ’еәҸпјҲеҰӮпјҡз”ЁжҲ·пјүпјҢеҸҲеҫ—йҮҚж–°з”іжҳҺдёҖдёӘжҺҘеҸЈгҖӮжңүдёҖдёӘеҠһжі•е°ұжҳҜеҸҜд»Ҙз”ЁJavaзҡ„жіӣеһӢеҜ№е®ғеҒҡиҝӣдёҖжӯҘзҡ„жҠҪиұЎпјҡ

public interface Predicate<T> {
    public boolean test(T t);
}

зҺ°еңЁдҪ еҸҜд»ҘжҠҠfilter()ж–№жі•з”ЁеңЁд»»дҪ•еҜ№иұЎзҡ„иҝҮж»ӨдёӯгҖӮ

LambdaиЎЁиҫҫејҸ

иҷҪз„¶жҲ‘д»¬еҜ№filter()ж–№жі•иҝӣиЎҢйҮҚжһ„пјҢе№¶жҠҪиұЎдәҶPredicateжҺҘеҸЈдҪңдёәиҝҮж»Өзҡ„жқЎд»¶пјҢдҪҶе®һйҷ…дёҠиҝҳйңҖиҰҒзј–еҶҷеҫҲеӨҡеҶ…йғЁзұ»жқҘе®һзҺ°PredicateжҺҘеҸЈгҖӮдҪҝз”ЁеҶ…йғЁзұ»зҡ„ж–№ејҸе®һзҺ°PredicateжҺҘеҸЈжңүеҫҲеӨҡзјәзӮ№пјҡйҰ–е…ҲжҳҜд»Јз Ғжҳҫеҫ—иҮғиӮҝдёҚе ӘпјҢеҸҜиҜ»жҖ§е·®пјӣе…¶ж¬ЎпјҢеҰӮжһңжҹҗдёӘеұҖйғЁеҸҳйҮҸиў«еҶ…йғЁзұ»дҪҝз”ЁпјҢиҝҷдёӘеҸҳйҮҸеҝ…йЎ»дҪҝз”Ёfinalе…ій”®еӯ—дҝ®йҘ°гҖӮеңЁJava 8дёӯпјҢдҪҝз”ЁLambdaиЎЁиҫҫејҸеҸҜд»ҘеҜ№еҶ…йғЁзұ»иҝӣдёҖжӯҘз®ҖеҢ–пјҡ

// ж №жҚ®дҪңиҖ…иҝҮж»Ө
List<Book> result = filter(books, book -> "еј дёү".equals(book.getAuthor()));

// ж №жҚ®еӣҫд№Ұд»·ж јиҝҮж»Ө
List<Book> result = filter(books, book -> 100 > book.getPrice());

дҪҝз”ЁLambdaд»…д»…з”ЁдёҖиЎҢд»Јз Ғе°ұеҜ№еҶ…йғЁзұ»иҝӣиЎҢдәҶиҪ¬еҢ–пјҢиҖҢдё”д»Јз ҒеҸҳеҫ—жӣҙеҠ жё…жҷ°еҸҜиҜ»гҖӮе…¶дёӯbook -> "еј дёү".equals(book.getAuthor())е’Ңbook -> 100 > book.getPrice()е°ұжҳҜжҲ‘д»¬жҺҘдёӢжқҘиҰҒз ”з©¶зҡ„LambdaиЎЁиҫҫејҸгҖӮ

LambdaиЎЁиҫҫејҸжҳҜд»Җд№Ҳ

LambdaиЎЁиҫҫејҸпјҲlambda expressionпјүжҳҜдёҖдёӘеҢҝеҗҚеҮҪж•°пјҢз”ұж•°еӯҰдёӯзҡ„О»жј”з®—иҖҢеҫ—еҗҚгҖӮеңЁJava 8дёӯеҸҜд»ҘжҠҠLambdaиЎЁиҫҫејҸзҗҶи§ЈдёәеҢҝеҗҚеҮҪж•°пјҢе®ғжІЎжңүеҗҚз§°пјҢдҪҶжҳҜжңүеҸӮж•°еҲ—иЎЁгҖҒеҮҪж•°дё»дҪ“гҖҒиҝ”еӣһзұ»еһӢзӯүгҖӮ

LambdaиЎЁиҫҫејҸзҡ„иҜӯжі•еҰӮдёӢпјҡ

(parameters) -> { statements; }

дёәд»Җд№ҲиҰҒдҪҝз”ЁLambdaиЎЁиҫҫејҸпјҹеүҚйқўдҪ д№ҹзңӢеҲ°дәҶпјҢеңЁJavaдёӯдҪҝз”ЁеҶ…йғЁзұ»жҳҫеҫ—еҚҒеҲҶеҶ—й•ҝпјҢиҰҒзј–еҶҷеҫҲеӨҡж ·жқҝд»Јз ҒпјҢLambdaиЎЁиҫҫејҸжӯЈжҳҜдёәдәҶз®ҖеҢ–иҝҷдәӣжӯҘйӘӨеҮәзҺ°зҡ„пјҢе®ғдҪҝд»Јз ҒеҸҳеҫ—жё…жҷ°жҳ“жҮӮгҖӮ

еҰӮдҪ•дҪҝз”ЁLambdaиЎЁиҫҫејҸ

LambdaиЎЁиҫҫејҸжҳҜдёәдәҶз®ҖеҢ–еҶ…йғЁзұ»зҡ„пјҢдҪ еҸҜд»ҘжҠҠе®ғеҪ“жҲҗжҳҜеҶ…йғЁзұ»зҡ„дёҖз§Қз®ҖеҶҷж–№ејҸпјҢеҸӘиҰҒжҳҜжңүеҶ…йғЁзұ»зҡ„д»Јз Ғеқ—пјҢйғҪеҸҜд»ҘиҪ¬еҢ–жҲҗLambdaиЎЁиҫҫејҸпјҡ

// ComparatorжҺ’еәҸ
List<Integer> list = Arrays.asList(3, 1, 4, 5, 2);
list.sort(new Comparator<Integer>() {
    @Override
    public int compare(Integer o1, Integer o2) {
        return o1.compareTo(o2);
    }
});

// дҪҝз”ЁLambdaиЎЁиҫҫејҸз®ҖеҢ–
list.sort((o1, o2) -> o1.compareTo(o2));
// Runnableд»Јз Ғеқ—
Thread thread = new Thread(new Runnable() {
    @Override
    public void run() {
        System.out.println("Hello Man!");
    }
});

// дҪҝз”ЁLambdaиЎЁиҫҫејҸз®ҖеҢ–
Thread thread = new Thread(() -> System.out.println("Hello Man!"));

еҸҜд»ҘзңӢеҮәпјҢеҸӘиҰҒжҳҜеҶ…йғЁзұ»зҡ„д»Јз Ғеқ—пјҢе°ұеҸҜд»ҘдҪҝз”ЁLambdaиЎЁиҫҫејҸз®ҖеҢ–пјҢе№¶дё”з®ҖеҢ–еҗҺзҡ„д»Јз Ғжё…жҷ°жҳ“жҮӮгҖӮз”ҡиҮіпјҢComparatorжҺ’еәҸзҡ„LambdaиЎЁиҫҫејҸиҝҳеҸҜд»ҘиҝӣдёҖжӯҘз®ҖеҢ–пјҡ

list.sort(Integer::compareTo);

иҝҷз§ҚеҶҷжі•иў«з§°дёә ж–№жі•еј•з”ЁпјҢж–№жі•еј•з”ЁжҳҜLambdaиЎЁиҫҫејҸзҡ„з®ҖдҫҝеҶҷжі•гҖӮеҰӮжһңдҪ зҡ„LambdaиЎЁиҫҫејҸеҸӘжҳҜи°ғз”ЁиҝҷдёӘж–№жі•пјҢжңҖеҘҪдҪҝз”ЁеҗҚз§°и°ғз”ЁпјҢиҖҢдёҚжҳҜжҸҸиҝ°еҰӮдҪ•и°ғз”ЁпјҢиҝҷж ·еҸҜд»ҘжҸҗй«ҳд»Јз Ғзҡ„еҸҜиҜ»жҖ§гҖӮ

ж–№жі•еј•з”ЁдҪҝз”Ё::еҲҶйҡ”з¬ҰпјҢеҲҶйҡ”з¬Ұзҡ„еүҚеҚҠйғЁеҲҶиЎЁзӨәеј•з”Ёзұ»еһӢпјҢеҗҺйқўеҚҠйғЁеҲҶиЎЁзӨәеј•з”Ёзҡ„ж–№жі•еҗҚз§°гҖӮдҫӢеҰӮпјҡInteger::compareToиЎЁзӨәеј•з”Ёзұ»еһӢдёәIntegerпјҢеј•з”ЁеҗҚз§°дёәcompareToзҡ„ж–№жі•гҖӮ

зұ»дјјдҪҝз”Ёж–№жі•еј•з”Ёзҡ„дҫӢеӯҗиҝҳжңүжү“еҚ°йӣҶеҗҲдёӯзҡ„е…ғзҙ еҲ°жҺ§еҲ¶еҸ°дёӯпјҡ

list.forEach(System.out::println);

еҮҪж•°ејҸжҺҘеҸЈ

еҰӮжһңдҪ зҡ„еҘҪеҘҮеҝғдҪҝдҪ зҝ»зңӢRunnableжҺҘеҸЈжәҗд»Јз ҒпјҢдҪ дјҡеҸ‘зҺ°иҜҘжҺҘеҸЈиў«дёҖдёӘ@FunctionalInterfaceзҡ„жіЁи§Јдҝ®йҘ°пјҢиҝҷжҳҜJava 8дёӯж·»еҠ зҡ„ж–°жіЁи§ЈпјҢз”ЁдәҺиЎЁзӨә еҮҪж•°ејҸжҺҘеҸЈгҖӮ

еҮҪж•°ејҸжҺҘеҸЈеҸҲжҳҜд»Җд№Ҳй¬јпјҹеңЁJava 8дёӯпјҢжҠҠйӮЈдәӣд»…жңүдёҖдёӘжҠҪиұЎж–№жі•зҡ„жҺҘеҸЈз§°дёәеҮҪж•°ејҸжҺҘеҸЈгҖӮеҰӮжһңдёҖдёӘжҺҘеҸЈиў«@FunctionalInterfaceжіЁи§Јж ҮжіЁпјҢиЎЁзӨәиҝҷдёӘжҺҘеҸЈиў«и®ҫи®ЎжҲҗеҮҪж•°ејҸжҺҘеҸЈпјҢеҸӘиғҪжңүдёҖдёӘжҠҪиұЎж–№жі•пјҢеҰӮжһңдҪ ж·»еҠ еӨҡдёӘжҠҪиұЎж–№жі•пјҢзј–иҜ‘ж—¶дјҡжҸҗзӨәвҖңMultiple non-overriding abstract methods found in interface XXXвҖқд№Ӣзұ»зҡ„й”ҷиҜҜгҖӮ

еҮҪж•°ејҸж–№жі•еҸҲиғҪеҒҡд»Җд№ҲпјҹJava8е…Ғи®ёдҪ д»ҘLambdaиЎЁиҫҫејҸзҡ„ж–№ејҸдёәеҮҪж•°ејҸжҺҘеҸЈжҸҗдҫӣе®һзҺ°пјҢйҖҡдҝ—зҡ„иҜҙпјҢдҪ еҸҜд»Ҙе°Ҷж•ҙдёӘLambdaиЎЁиҫҫејҸдҪңдёәжҺҘеҸЈзҡ„е®һзҺ°зұ»гҖӮ

йҷӨдәҶRunnableд№ӢеӨ–пјҢJava 8дёӯеҶ…зҪ®дәҶи®ёеӨҡеҮҪж•°ејҸжҺҘеҸЈдҫӣејҖеҸ‘иҖ…дҪҝз”ЁпјҢиҝҷдәӣжҺҘеҸЈдҪҚдәҺjava.util.functionеҢ…дёӯпјҢжҲ‘д»¬д№ӢеүҚдҪҝз”Ёзҡ„PredicateжҺҘеҸЈпјҢе·Із»Ҹиў«еҢ…еҗ«еңЁиҝҷдёӘеҢ…еҶ…пјҢд»–д»¬еҲҶеҲ«дёәPredicateгҖҒConsumerе’ҢFunctionпјҢз”ұдәҺжҲ‘д»¬е·Із»ҸеңЁд№ӢеүҚзҡ„еӣҫд№ҰиҝҮж»Өзҡ„дҫӢеӯҗдёӯд»Ӣз»ҚдәҶPredicateзҡ„з”Ёжі•пјҢжүҖд»ҘжҺҘдёӢжқҘдё»иҰҒд»Ӣз»ҚConsumerе’ҢFunctionзҡ„з”Ёжі•гҖӮ

Consumer

java.util.function.Consumer<T>е®ҡд№үдәҶдёҖдёӘеҗҚеҸ«accept()зҡ„жҠҪиұЎж–№жі•пјҢе®ғжҺҘеҸ—жіӣеһӢTзҡ„еҜ№иұЎпјҢжІЎжңүиҝ”еӣһпјҲvoidпјүгҖӮеҰӮжһңдҪ йңҖиҰҒи®ҝй—®зұ»еһӢTзҡ„еҜ№иұЎпјҢе№¶еҜ№е…¶жү§иЎҢжҹҗдәӣж“ҚдҪңпјҢе°ұеҸҜд»ҘдҪҝз”ЁиҝҷдёӘжҺҘеҸЈгҖӮжҜ”еҰӮпјҢдҪ еҸҜд»Ҙз”Ёе®ғжқҘеҲӣе»әдёҖдёӘforEach()ж–№жі•пјҢжҺҘеҸ—дёҖдёӘйӣҶеҗҲпјҢе№¶еҜ№йӣҶеҗҲдёӯжҜҸдёӘе…ғзҙ жү§иЎҢж“ҚдҪңпјҡ

@FunctionalInterface
public interface Consumer<T> {
    void accept(T t);
}

public static <T> void forEach(List<T> list, Consumer<T> consumer) {
    for(T t: list){
        consumer.accept(t);
    }
}

public static void main(String[] args) {
    List<String> list = Arrays.asList("A", "B", "C", "D");
    forEach(list, str -> System.out.println(str));
    // д№ҹеҸҜд»ҘеҶҷжҲҗ
    forEach(list, System.out::println);
}

Function

java.util.function.Function<T, R>жҺҘеҸЈе®ҡд№үдәҶдёҖдёӘеҸ«дҪңapply()зҡ„ж–№жі•пјҢе®ғжҺҘеҸ—дёҖдёӘжіӣеһӢTзҡ„еҜ№иұЎпјҢе№¶иҝ”еӣһдёҖдёӘжіӣеһӢRзҡ„еҜ№иұЎгҖӮеҰӮжһңдҪ йңҖиҰҒе®ҡд№үдёҖдёӘLambdaпјҢе°Ҷиҫ“е…ҘеҜ№иұЎзҡ„дҝЎжҒҜжҳ е°„еҲ°иҫ“еҮәпјҢе°ұеҸҜд»ҘдҪҝз”ЁиҝҷдёӘжҺҘеҸЈгҖӮжҜ”еҰӮпјҢжҲ‘д»¬йңҖиҰҒи®Ўз®—дёҖдёӘеӣҫд№ҰйӣҶеҗҲдёӯжҜҸжң¬д№Ұзҡ„дҪңиҖ…еҗҚз§°жңүеҮ дёӘжұүеӯ—пјҲеҒҮи®ҫиҝҷдәӣд№Ұзҡ„дҪңиҖ…йғҪжҳҜдёӯеӣҪдәәпјүпјҡ

@FunctionalInterface
public interface Function<T, R> {
    R apply(T t);
}

public static <T, R> List<R> map(List<T> list, Function<T, R> f) {
    List<R> result = new ArrayList<>();
    for(T s: list){
        result.add(f.apply(s));
    }
    return result;
}

public static void main(String[] args) {
    List<Book> books = Arrays.asList(
        new Book("еј дёү", 99.00D),
        new Book("жқҺеӣӣ", 59.00D),
        new Book("зҺӢиҖҒдә”", 59.00D)
    );
    List<Integer> results = map(books, book -> book.getAuthor().length());
}
